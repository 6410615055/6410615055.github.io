# เป้าหมายคืออาการป่วยมะเร็งปลอด
X = df.drop('LUNG_CANCER', axis=1)  # ลบคอลัมน์ target ออก
Y = df['LUNG_CANCER']  # เป้าหมาย (target)

# แบ่งข้อมูลเป็นชุดฝึก (train) และชุดทดสอบ (test) โดย 30% เป็น test
X_train, X_test, Y_train, Y_test = train_test_split(X, Y, test_size=0.3, random_state=42)

#แสดงขนาดของชุดข้อมูลที่เราได้ทำการแบ่ง
